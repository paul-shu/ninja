00398 bool Builder::Build(string* err) {
00399   if (!plan_.more_to_do()) {
00400     *err = "no work to do";
00401     return true;
00402   }
00403 
00404   status_->PlanHasTotalEdges(plan_.command_edge_count());
00405   int pending_commands = 0;
00406   int failures_allowed = config_.swallow_failures;
00407 
00408   // This main loop runs the entire build process.
00409   // It is structured like this:
00410   // First, we attempt to start as many commands as allowed by the
00411   // command runner.
00412   // Second, we attempt to wait for / reap the next finished command.
00413   // If we can do neither of those, the build is stuck, and we report
00414   // an error.
00415   while (plan_.more_to_do()) {
00416     // See if we can start any more commands.
00417     if (command_runner_->CanRunMore()) {
00418       if (Edge* edge = plan_.FindWork()) {
00419         if (!StartEdge(edge, err))
00420           return false;
00421 
00422         if (edge->is_phony())
00423           FinishEdge(edge, true, "");
00424         else
00425           ++pending_commands;
00426 
00427         // We made some progress; go back to the main loop.
00428         continue;
00429       }
00430     }
00431 
00432     // See if we can reap any finished commands.
00433     if (pending_commands) {
00434       bool success;
00435       string output;
00436       Edge* edge;
00437       if ((edge = command_runner_->WaitForCommand(&success, &output))) {
00438         --pending_commands;
00439         FinishEdge(edge, success, output);
00440         if (!success) {
00441           if (--failures_allowed < 0) {
00442             if (config_.swallow_failures > 0)
00443               *err = "subcommands failed";
00444             else
00445               *err = "subcommand failed";
00446             return false;
00447           }
00448         }
00449 
00450         // We made some progress; start the main loop over.
00451         continue;
00452       }
00453     }
00454 
00455     // If we get here, we can neither enqueue new commands nor are any running.
00456     if (pending_commands) {
00457       *err = "stuck: pending commands but none to wait for? [this is a bug]";
00458       return false;
00459     }

00461     // If we get here, we cannot make any more progress.
00462     if (failures_allowed < config_.swallow_failures) {
00463       *err = "cannot make progress due to previous errors";
00464       return false;
00465     } else {
00466       *err = "stuck [this is a bug]";
00467       return false;
00468     }
00469   }
00470 
00471   return true;
00472 }
