00074 int GuessParallelism() {
00075   int processors = 0;
00076 
00077 #if defined(linux)
00078   processors = get_nprocs();
00079 #elif defined(__APPLE__) || defined(__FreeBSD__)
00080   size_t processors_size = sizeof(processors);
00081   int name[] = {CTL_HW, HW_NCPU};
00082   if (sysctl(name, sizeof(name) / sizeof(int),
00083              &processors, &processors_size,
00084              NULL, 0) < 0) {
00085     processors = 1;
00086   }
00087 #elif defined(WIN32)
00088   SYSTEM_INFO info;
00089   GetSystemInfo(&info);
00090   processors = info.dwNumberOfProcessors;
00091 #endif
00092 
00093   switch (processors) {
00094   case 0:
00095   case 1:
00096     return 2;
00097   case 2:
00098     return 3;
00099   default:
00100     return processors + 2;
00101   }
00102 }
