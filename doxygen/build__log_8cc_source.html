00026 // Implementation details:
00027 // Each run's log appends to the log file.

00148       // int time_ms = atoi(start);

00150     } else {
00151       // In v2 we log the start time and the end time.
00152       start_time = atoi(start);

00154 
00155       char* end = strchr(start, ' ');
00156       if (!end)
00157         continue;
00158       *end = 0;
00159       end_time = atoi(start);
00160       start = end + 1;
00161     }
00162 
00163     end = strchr(start, ' ');
00164     if (!end)
00165       continue;
00166     string output = string(start, end - start);
00167 
00168     start = end + 1;
00169     end = strchr(start, '\n');
00170     if (!end)
00171       continue;
00172 
00173     LogEntry* entry;
00174     Log::iterator i = log_.find(output);
00175     if (i != log_.end()) {
00176       entry = i->second;
00177     } else {
00178       entry = new LogEntry;
00179       log_.insert(make_pair(output, entry));
00180       ++unique_entry_count;
00181     }
00182     ++total_entry_count;
00183 
00184     entry->output = output;
00185     entry->start_time = start_time;
00186     entry->end_time = end_time;
00187     entry->command = string(start, end - start);
00188   }
00189 
00190   // Decide whether it's time to rebuild the log:
00191   // - if we're upgrading versions
00192   // - if it's getting large
00193   int kMinCompactionEntryCount = 100;
00194   int kCompactionRatio = 3;
00195   if (log_version < kCurrentVersion) {
00196     needs_recompaction_ = true;
00197   } else if (total_entry_count > kMinCompactionEntryCount &&
00198              total_entry_count > unique_entry_count * kCompactionRatio) {
00199     needs_recompaction_ = true;
00200   }
00201 
00202   fclose(file);
00203 
00204   return true;
