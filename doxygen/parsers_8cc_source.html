00073 void Tokenizer::SkipWhitespace(bool newline) {
00074   if (token_.type_ == Token::NEWLINE && newline)
00075     Newline(NULL);
00076 
00077   const char kContinuation = makefile_flavor_ ? '\\' : '$';
00078 
00079   while (cur_ < end_) {
00080     if (*cur_ == ' ') {
00081       ++cur_;
00082     } else if (newline && *cur_ == '\n') {
00083       Newline(NULL);
00084     } else if (*cur_ == kContinuation && cur_ + 1 < end_ && cur_[1] == '\n') {
00085       ++cur_; ++cur_;
00086     } else if (*cur_ == '#' && cur_ == cur_line_) {
00087       while (cur_ < end_ && *cur_ != '\n')
00088         ++cur_;
00089       if (cur_ < end_ && *cur_ == '\n')
00090         ++cur_;
00091       cur_line_ = cur_;
00092     } else {
00093       break;
00094     }
00095   }

00140 // A note on backslashes in Makefiles, from reading the docs:
00141 // Backslash-newline is the line continuation character.
00142 // Backslash-# escapes a # (otherwise meaningful as a comment start).
00143 // Backslash-% escapes a % (otherwise meaningful as a special).
00144 // Finally, quoting the GNU manual, "Backslashes that are not in danger
00145 // of quoting ‘%’ characters go unmolested."
00146 // How do you end a line with a backslash?  The netbsd Make docs suggest
00147 // reading the result of a shell command echoing a backslash!
00148 //
00149 // Rather than implement the above, we do the simpler thing here.
00150 // If anyone actually has depfiles that rely on the more complicated
00151 // behavior we can adjust this.
00152 bool Tokenizer::ReadToNewline(string *text, string* err, size_t max_length) {
00153   // XXX token_.clear();
00154   const char kContinuation = makefile_flavor_ ? '\\' : '$';
00155   while (cur_ < end_ && *cur_ != '\n') {
00156     if (*cur_ == kContinuation) {
00157       // Might be a line continuation; peek ahead to check.
00158       if (cur_ + 1 >= end_)
00159         return Error("unexpected eof", err);
00160       if (*(cur_ + 1) == '\n') {
00161         // Let SkipWhitespace handle the continuation logic.
00162         SkipWhitespace();
00163         continue;
00164       }
00165 
00166       // Otherwise, just treat it like a normal character.
00167       text->push_back(*cur_);
00168       ++cur_;
00169     } else {
00170       text->push_back(*cur_);
00171       ++cur_;
00172     }
00173     if (text->size() >= max_length) {
00174       token_.pos_ = cur_;
00175       return false;
00176     }
00177   }
00178   return Newline(err);
00179 }

00226     ++cur_;

00229   }

00232 
00233   if (token_.type_ == Token::NONE) {

00282 bool ManifestParser::Parse(const string& input, string* err) {
00283   tokenizer_.Start(input.data(), input.data() + input.size());
00284 
00285   tokenizer_.SkipWhitespace(true);
00286 
00287   while (tokenizer_.token().type_ != Token::TEOF) {
00288     switch (tokenizer_.PeekToken()) {
00289       case Token::IDENT: {
00290         const Token& token = tokenizer_.token();
00291         int len = token.end_ - token.pos_;
00292         if (len == 4 && memcmp(token.pos_, "rule", 4) == 0) {
00293           if (!ParseRule(err))
00294             return false;
00295         } else if (len == 5 && memcmp(token.pos_, "build", 5) == 0) {
00296           if (!ParseEdge(err))
00297             return false;
00298         } else if (len == 7 && memcmp(token.pos_, "default", 7) == 0) {
00299           if (!ParseDefaults(err))
00300             return false;
00301         } else if ((len == 7 && memcmp(token.pos_, "include", 7) == 0) ||
00302                    (len == 8 && memcmp(token.pos_, "subninja", 8) == 0)) {
00303           if (!ParseFileInclude(err))
00304             return false;
00305         } else {
00306           string name, value;
00307           if (!ParseLet(&name, &value, err))
00308             return false;
00309           env_->AddBinding(name, value);
00310         }
00311         break;
00312       }
00313       case Token::TEOF:
00314         continue;
00315       default:
00316         return tokenizer_.Error("unhandled " + tokenizer_.token().AsString(), err);
00317     }
00318     tokenizer_.SkipWhitespace(true);
00319   }
00320 
00321   return true;
