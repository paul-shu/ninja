00031 bool Edge::RecomputeDirty(State* state, DiskInterface* disk_interface,
00032                           string* err) {
00033   bool dirty = false;
00034 
00035   if (!rule_->depfile_.empty()) {
00036     if (!LoadDepFile(state, disk_interface, err))
00037       return false;
00038   }
00039 
00040   time_t most_recent_input = 1;
00041   for (vector<Node*>::iterator i = inputs_.begin(); i != inputs_.end(); ++i) {
00042     if ((*i)->file_->StatIfNecessary(disk_interface)) {
00043       if (Edge* edge = (*i)->in_edge_) {
00044         if (!edge->RecomputeDirty(state, disk_interface, err))
00045           return false;
00046       } else {
00047         // This input has no in-edge; it is dirty if it is missing.
00048         (*i)->dirty_ = !(*i)->file_->exists();
00049       }
00050     }
00051 
00052     if (is_order_only(i - inputs_.begin())) {
00053       // Order-only deps only make us dirty if they're missing.
00054       if (!(*i)->file_->exists())
00055         dirty = true;
00056       continue;
00057     }
00058 
00059     // If a regular input is dirty (or missing), we're dirty.
00060     // Otherwise consider mtime.
00061     if ((*i)->dirty_) {
00062       dirty = true;
00063     } else {
00064       if ((*i)->file_->mtime_ > most_recent_input)
00065         most_recent_input = (*i)->file_->mtime_;
00066     }
00067   }
00068 
00069   string command = EvaluateCommand();
00070 
00071   assert(!outputs_.empty());
00072   for (vector<Node*>::iterator i = outputs_.begin(); i != outputs_.end(); ++i) {
00073     // We may have other outputs, that our input-recursive traversal hasn't hit
00074     // yet (or never will).  Stat them if we haven't already.
00075     (*i)->file_->StatIfNecessary(disk_interface);
00076 
00077     // Output is dirty if we're dirty, we're missing the output,
00078     // or if it's older than the most recent input mtime.
00079     if (dirty || !(*i)->file_->exists() ||
00080         (*i)->file_->mtime_ < most_recent_input) {
00081       (*i)->dirty_ = true;
00082     } else {
00083       // May also be dirty due to the command changing since the last build.
00084       BuildLog::LogEntry* entry;
00085       if (state->build_log_ &&
00086           (entry = state->build_log_->LookupByOutput((*i)->file_->path_))) {
00087         if (command != entry->command)
00088           (*i)->dirty_ = true;
00089       }
00090     }
00091   }
00092   return true;
00093 }

00135                        string* err) {

00148     *err = path + ": " + makefile_err;

00153   if (outputs_.size() != 1) {
00154     *err = "expected only one output";
00155     return false;
00156   }
00157   if (outputs_[0]->file_->path_ != makefile.out_) {
00158     *err = "expected makefile to mention '" + outputs_[0]->file_->path_ + "', "
00159            "got '" + makefile.out_ + "'";
00160     return false;
00161   }
00162 
00163   // Add all its in-edges.
00164   for (vector<string>::iterator i = makefile.ins_.begin();
00165        i != makefile.ins_.end(); ++i) {
00166     CanonicalizePath(&*i);
00167 
00168     Node* node = state->GetNode(*i);
00169     inputs_.insert(inputs_.end() - order_only_deps_, node);
00170     node->out_edges_.push_back(this);
00171     ++implicit_deps_;
00172 
00173     // If we don't have a edge that generates this input already,
00174     // create one; this makes us not abort if the input is missing,
00175     // but instead will rebuild in that circumstance.
00176     if (!node->in_edge_) {
00177       Edge* phony_edge = state->AddEdge(&State::kPhonyRule);

00179       node->in_edge_ = phony_edge;

00182   }
