00246 bool BuildTest::StartCommand(Edge* edge) {
00247   assert(!last_command_);
00248   commands_ran_.push_back(edge->EvaluateCommand());
00249   if (edge->rule_->name_ == "cat" || edge->rule_->name_ == "cc" ||
00250         edge->rule_->name_ == "touch") {
00251     for (vector<Node*>::iterator out = edge->outputs_.begin();

00253       (*out)->file_->mtime_ = now_;

00257     // Don't do anything.
